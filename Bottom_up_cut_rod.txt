int cutrod(int price[],int size){

if (size <= 0) 
     return 0; 
   int max_val = INT_MIN; 
  
 
   for (int i = 0; i<size; i++) 
         max_val = max(max_val, price[i] + cutrod(price, size-i-1)); 
  
   return max_val; 
}

int max(int a, int b) { return (a > b)? a : b;}
